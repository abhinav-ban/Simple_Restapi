import React, { useState } from "react";
import "./App.css";

function Use(){
    const[m,setM]=useState("")
    const handleGet = async () => {
    const response = await fetch("http://127.0.0.1:5000/get");
    const data = await response.json();
    setM(JSON.stringify(data));
  };
   const handlePost = async () => {
    const response = await fetch("http://127.0.0.1:5000/post", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ value: "New Data" }),
    });
    const data = await response.json();
    setM(JSON.stringify(data));
  };
}

function App(){
    return(
        <div className="parent">
            <header className="header">HANNITY</header>
            <footer className="footer">
                <button className="daylight" onClick={Use.handleGet}>GET</button>
                <button className="signup" onClick={Use.handlePost}>POST</button>
            </footer>
        </div>
    
);
}
export default App;
